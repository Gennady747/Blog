recent_themes = []
recent_hashtags = []

def get_unique_story():
    """Берём историю, которую не использовали недавно"""
    global recent_themes
    
    available = [s for s in STORIES if s["theme"] not in recent_themes]
    if not available:
        recent_themes = []
        available = STORIES
    
    story = random.choice(available)
    recent_themes.append(story["theme"])
    if len(recent_themes) > 15:  # Увеличил до 15 для 300+ историй
        recent_themes.pop(0)
    
    return story

def get_unique_hashtags():
    """Берём хэштеги, которые не использовали недавно"""
    global recent_hashtags
    
    available = [h for h in HASHTAGS if h not in recent_hashtags]
    if not available:
        recent_hashtags = []
        available = HASHTAGS
    
    tags = random.choice(available)
    recent_hashtags.append(tags)
    if len(recent_hashtags) > 5:
        recent_hashtags.pop(0)
    
    return tags

def format_post(story):
    """Форматируем пост: эмодзи + текст + хэштеги"""
    theme = story["theme"]
    text = story["text"]
    
    # Выбираем эмодзи по теме или случайные
    emojis = EMOJI_THEMES.get(theme, ["🌿", "💭", "✨"])
    emoji = random.choice(emojis)
    
    # Хэштеги
    hashtags = get_unique_hashtags()
    
    # Собираем пост
    post = f"{emoji} {text}\n\n{hashtags}"
    
    return post

def send_photo(photo_url, caption):
    url = f"https://api.telegram.org/bot{BOT_TOKEN}/sendPhoto"
    try:
        r = requests.post(url, json={
            "chat_id": CHANNEL_ID,
            "photo": photo_url,
            "caption": caption[:1024],
            "parse_mode": "HTML"
        }, timeout=30)
        
        if not r.json().get("ok", False):
            url_text = f"https://api.telegram.org/bot{BOT_TOKEN}/sendMessage"
            requests.post(url_text, json={
                "chat_id": CHANNEL_ID,
                "text": caption,
                "parse_mode": "HTML"
            })
        return True
    except Exception as e:
        print(f"   ❌ Ошибка: {e}")
        return False

def get_motivational_image():
    topics = ["motivation", "selfgrowth", "sunrise", "reflection", "nature", "thinking", "calm", "inspiration"]
    keyword = random.choice(topics)
    return f"https://loremflickr.com/800/600/{keyword}?lock={random.randint(1, 999999)}"
