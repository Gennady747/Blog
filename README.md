# Функция для отправки поста
def send_telegram_post(bot_token, chat_id, text):
    url = f"https://api.telegram.org/bot{bot_token}/sendMessage"
    payload = {
        "chat_id": chat_id,
        "text": text,
        "parse_mode": "HTML"
    }
    try:
        response = requests.post(url, json=payload)
        if response.status_code == 200:
            print(f"[{datetime.now().strftime('%H:%M:%S')}] Пост успешно отправлен!")
            return True
        else:
            print(f"Ошибка отправки: {response.text}")
            return False
    except Exception as e:
        print(f"Произошла ошибка: {e}")
        return False

def format_post(story):
    """Форматирует историю для красивого вида в Telegram"""
    header = f"<b>📌 Тема: {story['theme'].upper()}</b>\n\n"
    body = story['text']
    footer = f"\n\n<i>#история #мотивация #life</i>"
    return header + body + footer

def main():
    print(f"Бот запущен! Интервал постов: {POST_INTERVAL} сек.")
    print(f"Используется канал: {CHANNEL_ID}")
    
    # Чтобы не повторяться, перемешиваем список
    shuffled_stories = STORIES.copy()
    random.shuffle(shuffled_stories)
    
    index = 0
    while True:
        if index >= len(shuffled_stories):
            print("Все истории опубликованы. Начинаем заново и перемешиваем...")
            random.shuffle(shuffled_stories)
            index = 0
            
        story = shuffled_stories[index]
        post_text = format_post(story)
        
        if send_telegram_post(BOT_TOKEN, CHANNEL_ID, post_text):
            index += 1
            print(f"Следующий пост через {POST_INTERVAL} секунд...")
            time.sleep(POST_INTERVAL)
        else:
            print("Повторная попытка через 30 секунд...")
            time.sleep(30)

if __name__ == "__main__":
    main()
